سبحان ربي الأعلى . ثلاث مرات

--------------------------------------------------

أخرجه أهل السنن، وأحمد: أبو داود، برقم ٨٧٠، والترمذي، برقم ٢٦٢، والنسائي، برقم ١٠٠٧، وابن ماجه، برقم ٨٩٧، وأحمد، برقم، ٣٥١٤، وانظر: صحيح الترمذي، ١/٨٣


سبحانك اللهم ربنا وبحمدك، اللهم اغفر لي

--------------------------------------------------

البخاري، برقم، ٧٩٤، ومسلم، برقم ٤٨٤، وتقدم برقم ٣٤


سبوح، قدوس، رب الملائكة والروح

--------------------------------------------------

مسلم، ١/ ٥٣٣، برقم ٤٨٧، وأبو داود، برقم ٨٧٢، وتقدم برقم ٣٥


اللهم لك سجدت وبك آمنت، ولك أسلمت، سجد وجهي للذي خلقه، وصوره، وشق سمعه وبصره، تبارك الله أحسن الخالقين

--------------------------------------------------

مسلم، ١/ ٥٣٤، برقم ٧٧١، وغيره


سبحان ذي الجبروت، والملكوت، والكبرياء، والعظمة

--------------------------------------------------

أبو داود، ١/ ٢٣٠، برقم ٨٧٣، والنسائي، برقم ١١٣١، وأحمد، برقم ٢٣٩٨٠، وصححه الألباني في صحيح أبي داود، ١/ ١٦٦، وتقدم تخريجه برقم ٣٧


اللهم اغفر لي ذنبي كله: دقه وجله، وأوله وآخره، وعلانيته وسره

--------------------------------------------------

مسلم، ١/ ٢٣٠، برقم ٤٨٣
